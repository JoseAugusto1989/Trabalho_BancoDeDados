

CREATE TABLE Banco (
    Id Integer PRIMARY KEY,
    Nome Varchar(40),
    CNPJ Varchar(26)
);

CREATE TABLE Conta Bancaria (
    Id_Conta Integer PRIMARY KEY,
    CPF_Cliente Integer,
    Nome_Cliente Varchar(40),
    Saldo_Conta Double
);

CREATE TABLE Movimentos (
    Id_Cliente Integer PRIMARY KEY,
    Tipos_Movimentação Varchar(20),
    Valor Double
);
 
ALTER TABLE Conta Bancaria ADD CONSTRAINT FK_Conta Bancaria_2
    FOREIGN KEY (Id_Conta)
    REFERENCES Banco (Id);
 
ALTER TABLE Conta Bancaria ADD CONSTRAINT FK_Conta Bancaria_3
    FOREIGN KEY (CPF_Cliente)
    REFERENCES Movimentos (Id_Cliente);
